```go
package sqlite

import (
	"database/sql"
	"fmt"
	"os"
	
	_ "github.com/mattn/go-sqlite3"
)

// Init initializes the SQLite database with schema
func Init(dbPath string) (*sql.DB, error) {
	// Ensure data directory exists
	dir := "data"
	if err := os.MkdirAll(dir, 0755); err != nil {
		return nil, fmt.Errorf("failed to create data directory: %w", err)
	}
	
	// Open database
	db, err := sql.Open("sqlite3", dbPath)
	if err != nil {
		return nil, fmt.Errorf("failed to open database: %w", err)
	}
	
	// Execute schema
	schema := getSchema()
	_, err = db.Exec(schema)
	if err != nil {
		db.Close()
		return nil, fmt.Errorf("failed to execute schema: %w", err)
	}
	
	return db, nil
}
```

```go
package config

import (
	"fmt"
	"os"
	
	"gopkg.in/yaml.v3"
)

func Load(path string) (*Config, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("failed to read config file: %w", err)
	}
	
	var cfg Config
	err = yaml.Unmarshal(data, &cfg)
	if err != nil {
		return nil, fmt.Errorf("failed to parse config file: %w", err)
	}
	
	return &cfg, nil
}
```

```markdown
1. **Presentation Layer**: React frontend (TypeScript)
```

  ai/                - AI interaction
  monitor/           - Monitoring integration
  alert/             - Alert management
  remediation/       - Auto-remediation
  knowledge/         - Knowledge base
  audit/             - Audit logging
  storage/           - Data access layer
    sqlite/          - SQLite operations
    file/            - File storage operations
  grpc/              - gRPC services (Agent communication)
  k8s/               - Kubernetes client wrapper

configs/             - Configuration files
scripts/             - Deployment and setup scripts

tests/               - Integration tests
